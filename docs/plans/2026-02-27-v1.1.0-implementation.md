        uses: docker/metadata-action@v5
        with:
          images: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}
          tags: |
            type=raw,value=latest,enable=${{ github.ref == 'refs/heads/main' }}
            type=raw,value=dev,enable=${{ github.ref == 'refs/heads/dev' }}
            type=sha,prefix=sha-

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

// Run once to create the knowledge_base schema (or a custom schema).
// Usage:
//   node scripts/run-migration.js                          # creates knowledge_base
//   SCHEMA_NAME=knowledge_base_staging node scripts/run-migration.js  # creates staging schema

const { Pool } = require('pg');

const path = require('path');

const schemaName = process.env.SCHEMA_NAME || 'knowledge_base';

const pool = new Pool({
  connectionString: process.env.DATABASE_URL ||
    'postgresql://nocodb:nocodb2026@192.168.86.18:32775/nocodb'
});

async function runMigration() {
  let sql = fs.readFileSync(
    path.join(__dirname, 'migrate.sql'),
    'utf8'
  );

  // Replace schema name if overridden
  if (schemaName !== 'knowledge_base') {
    sql = sql.replace(/knowledge_base\b/g, schemaName);
  }

  console.log(`Running migration for schema: ${schemaName}`);
  try {
    await pool.query(sql);
    console.log(`✓ Migration complete — ${schemaName} schema created`);
  } catch (err) {
    console.error('✗ Migration failed:', err.message);
    process.exit(1);
  } finally {
    await pool.end();
  }
}

// Helper to flatten tree for ID lookup
function flattenTree(nodes) {
  return nodes.reduce((acc, node) => {
    acc.push(node);

// ── Tree query: all pages for a section ───────────────────
async function getPageTree(sectionId, role = 'viewer') {
  // Archived pages always hidden.
  // Draft pages hidden from viewers.
  const isPrivileged = role === 'admin' || role === 'editor';

  let whereClause = `WHERE section_id = $1 AND deleted_at IS NULL AND status != 'archived'`;
  if (!isPrivileged) {
    whereClause += ` AND status = 'published'`;
  }

  const res = await getPool().query(`
    SELECT id, parent_id, title, slug, status, template_type,
           created_by, sort_order, created_at, updated_at
    FROM knowledge_base.pages
    ${whereClause}
    ORDER BY sort_order, title
  `, [sectionId]);
  return buildTree(res.rows);
}

async function listAssets({ type, q } = {}) {
  let where = 'WHERE a.deleted_at IS NULL';
  const values = [];

  if (type) {
    values.push(type);
    where += ` AND a.type = $${values.length}`;
  }

  if (q) {
    values.push(`%${q.toLowerCase()}%`);
    where += ` AND (LOWER(a.title) LIKE $${values.length} OR LOWER(a.description) LIKE $${values.length})`;
  }

  const res = await getPool().query(
    `SELECT id, type, title, description, file_path, url, metadata, created_by, created_at, updated_at
     FROM knowledge_base.assets a ${where} ORDER BY updated_at DESC`,
    values
  );
  return res.rows;
}

// Standalone asset browser — lists all assets, filterable by type and search query.
'use strict';

function renderAssetsBrowser() {
  const panel = document.getElementById('assets-browser-panel');
  if (!panel) return;

  // Build filter bar
  const typeSelect = document.createElement('select');
  typeSelect.id = 'ab-type-filter';
  ASSET_TYPES.forEach(t => {
    const opt = document.createElement('option');
    opt.value = t.value;
    opt.textContent = t.label;
    typeSelect.appendChild(opt);
  });
  typeSelect.addEventListener('change', e => {
    currentType = e.target.value;
    loadAssets();

  searchInput.type = 'text';
  searchInput.id = 'ab-search';
  searchInput.placeholder = 'Search assets…';
  searchInput.addEventListener('input', e => {

      currentQuery = e.target.value.trim();
      loadAssets();
    }, 280);
  });

  const filterBar = document.createElement('div');
  filterBar.className = 'ab-filter-bar';
  filterBar.appendChild(searchInput);
  filterBar.appendChild(typeSelect);

  const list = document.createElement('div');
  list.id = 'ab-list';
  list.className = 'ab-list';

async function loadAssets() {
  const list = document.getElementById('ab-list');
  if (!list) return;

  list.textContent = '';
  const loading = document.createElement('p');
  loading.className = 'ab-loading';
  loading.textContent = 'Loading…';
  list.appendChild(loading);

  try {
    const params = new URLSearchParams();
    if (currentType) params.set('type', currentType);
    if (currentQuery) params.set('q', currentQuery);

    const assets = await api.get(`/assets?${params.toString()}`);
    list.textContent = '';

    if (assets.length === 0) {
      const empty = document.createElement('p');
      empty.className = 'ab-empty';
      empty.textContent = 'No assets found.';
      list.appendChild(empty);
      return;
    }

    assets.forEach(asset => {
      const item = document.createElement('div');
      item.className = 'ab-item';
      item.setAttribute('data-id', asset.id);

      const badge = document.createElement('span');
      badge.className = `ab-type-badge type-${asset.type}`;
      badge.textContent = asset.type;

      const titleEl = document.createElement('span');
      titleEl.className = 'ab-item-title';
      titleEl.textContent = asset.title;

      const meta = document.createElement('span');
      meta.className = 'ab-item-meta';
      meta.textContent = new Date(asset.updated_at).toLocaleDateString();

      item.appendChild(badge);
      item.appendChild(titleEl);
      item.appendChild(meta);

      item.addEventListener('click', () => openAssetDetail(asset.id));
      list.appendChild(item);
    });
  } catch (err) {
    list.textContent = '';
    const errEl = document.createElement('p');
    errEl.className = 'ab-error';
    errEl.textContent = 'Failed to load assets.';
    list.appendChild(errEl);
    showToast('Failed to load assets', 'error');
  }
}

# Usage: bash scripts/setup-labels.sh <repo>

   Expected: all tests pass. Do not proceed if any test fails.

2. **Smoke test staging**
   Open browser → https://{project}-staging.ss-42.com
   Verify: login works, key features work, no console errors.

## Release steps

3. **Determine version number**
   ```bash
   git describe --tags --abbrev=0
   ```
   Increment following SemVer:
   - Bug fix only → PATCH (v1.0.1)
   - New feature → MINOR (v1.1.0)
   - Breaking change → MAJOR (v2.0.0)

4. **Create PR: dev → main**
   ```bash
   gh pr create \
     --repo sspaynter/{project} \
     --base main \
     --head dev \
     --title "Release v{X.Y.Z}" \
     --body "## Changes in this release\n\nCloses #{issue numbers}\n\n## Smoke test\n- [ ] Staging verified at {project}-staging.ss-42.com"
   ```

5. **Merge the PR**
   ```bash
   gh pr merge --repo sspaynter/{project} --squash
   ```
   CI builds :latest → Watchtower deploys to production automatically.

6. **Cut git tag**
   ```bash
   git checkout main && git pull origin main
   git tag v{X.Y.Z}
   git push origin v{X.Y.Z}
   ```

7. **Create GitHub Release**
   ```bash
   gh release create v{X.Y.Z} \
     --repo sspaynter/{project} \
     --title "v{X.Y.Z}" \
     --notes "## What's changed\n\n{summary of changes}\n\n**Full changelog:** see CHANGELOG.md"
   ```

8. **Update CHANGELOG.md**
   Add entry at top of CHANGELOG.md:
   ```markdown
   ## [v{X.Y.Z}] — {YYYY-MM-DD}
   ### Added
   - {new features}
   ### Fixed
   - {bug fixes}
   ### Changed
   - {other changes}
   ```

9. **Create KB release notes page**
   - Open kb.ss-42.com → {Project} workspace → Releases
   - Create new page using `release-notes` template
   - Title: "v{X.Y.Z} — {YYYY-MM-DD}"
   - Fill in: what is new, what changed, what was fixed

10. **Close GitHub milestone**
    ```bash
    gh api repos/sspaynter/{project}/milestones \
      --jq '.[] | select(.title=="v{X.Y.Z}") | .number'
    # Then close it:
    gh api repos/sspaynter/{project}/milestones/{N} -X PATCH -f state=closed
    ```

11. **Update KB project overview page**
    Open the project's Overview page in KB. Update the "Current version" and "Current status" fields.

## After release

Merge main back into dev to keep branches in sync:
```bash
git checkout dev
git merge main
git push origin dev
```
```
